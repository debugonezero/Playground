async function handleInjection() { const log = document.getElementById('log'); try { const raw = await navigator.clipboard.readText(); const data = JSON.parse(raw); // The Packet Unpacker // 1. Route to GPU if (data.wgsl && window.device) { const mod = window.device.createShaderModule({ code: data.wgsl }); window.pipeline = window.device.createRenderPipeline({ layout: 'auto', vertex: { module: mod, entryPoint: 'vs_main' }, fragment: { module: mod, entryPoint: 'fs_main', targets: [{ format: window.format }] }, primitive: { topology: 'point-list' } }); window.bindGroup = window.device.createBindGroup({ layout: window.pipeline.getBindGroupLayout(0), entries: [{ binding: 0, resource: { buffer: window.uniformBuffer } }] }); } // 2. Route to UI/DOM if (data.html) { const container = document.getElementById('render-target'); const wrapper = document.createElement('div'); wrapper.innerHTML = data.html; container.appendChild(wrapper); } // 3. Route to Logic if (data.js) { eval(data.js); } log.innerText = "PACKET_UNPACKED: SUCCESS"; } catch (err) { log.innerText = "PACKET_ERROR: CHECK_FORMAT"; console.error(err); } }
